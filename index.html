قسم الأمن السيبراني - المرحلة الأولى

المادة: برمجة (C++)

اسم الطالب: عبدالله احمد مطلك

الموضوع: حلول تمارين المصفوفات (Arrays)

السؤال الأول: إيجاد القيم العظمى في أعمدة المصفوفة (4x4)

برنامج يقوم بقراءة مصفوفة ثنائية الأبعاد، ثم يستخرج أكبر قيمة في كل عمود ويخزنها في مصفوفة أحادية.

#include <iostream>
using namespace std;

int main() {
    int dataGrid[4][4], maxInCols[4];
    
    for (int r = 0; r < 4; r++)
        for (int c = 0; c < 4; c++)
            cin >> dataGrid[r][c];

    for (int c = 0; c < 4; c++) {
        maxInCols[c] = dataGrid[0][c];
        for (int r = 1; r < 4; r++) {
            if (dataGrid[r][c] > maxInCols[c])
                maxInCols[c] = dataGrid[r][c];
        }
    }

    cout << "Maximums: ";
    for (int i = 0; i < 4; i++)
        cout << maxInCols[i] << " ";

    return 0;
}

السؤال الثاني: حساب متوسط الأعداد الفردية في مصفوفة محولة

تحويل مصفوفة ثنائية (3x3) إلى أحادية وحساب المتوسط الحسابي للأعداد الفردية فقط.

#include <iostream>
using namespace std;

int main() {
    int r = 3, c = 3, k = 0;
    int matrix[3][3], flatArr[9];
    int sumOdd = 0, countOdd = 0;

    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++) {
            cin >> matrix[i][j];
            flatArr[k++] = matrix[i][j];
        }
    }

    for (int i = 0; i < 9; i++) {
        if (flatArr[i] % 2 != 0) {
            sumOdd += flatArr[i];
            countOdd++;
        }
    }

    if (countOdd > 0)
        cout << "Average of Odd: " << (double)sumOdd / countOdd;
    else
        cout << "No odd numbers found.";

    return 0;
}

السؤال الثالث: فحص الأعداد الأولية في القطر الثانوي

برنامج مخصص لعد الأرقام الأولية الموجودة في عناصر القطر الثانوي للمصفوفة المربعة.

#include <iostream>
using namespace std;

int main() {
    int board[4][4], primeCount = 0;
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 4; j++)
            cin >> board[i][j];

    for (int i = 0; i < 4; i++) {
        int val = board[i][3-i];
        bool check = true;
        if (val < 2) check = false;
        for (int x = 2; x < val; x++)
            if (val % x == 0) check = false;
        if (check) primeCount++;
    }
    cout << "Count = " << primeCount;
    return 0;
}

السؤال الرابع: مجموع أعمدة المصفوفة

#include <iostream>
using namespace std;

int main() {
    int arr[4][4], colTotals[4] = {0};
    for (int r = 0; r < 4; r++)
        for (int c = 0; c < 4; c++)
            cin >> arr[r][c];

    for (int c = 0; c < 4; c++)
        for (int r = 0; r < 4; r++)
            colTotals[c] += arr[r][c];

    for (int i = 0; i < 4; i++)
        cout << colTotals[i] << " ";
    return 0;
}

السؤال الخامس: عكس مصفوفة أحادية

#include <iostream>
using namespace std;

int main() {
    int n, vec[100];
    cin >> n;
    for (int i = 0; i < n; i++) cin >> vec[i];

    for (int i = n - 1; i >= 0; i--)
        cout << vec[i] << " ";
    return 0;
}
